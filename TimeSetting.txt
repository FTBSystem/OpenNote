sudo apt remove systemd-timesyncd --purge #removed systemd-timesyncd

sudo apt install systemd-timesyncd -y #install systemd-timesyncd

systemctl start systemd-timesyncd.service #start systemd-timesyncd

sudo apt-get install util-linux-extra -y #install hwclock

sudo timedatectl set-ntp true  #use NTP

sudo timedatectl set-local-rtc 1 --adjust-system-clock #sync RTC/System clock clock

sudo hwclock -w #(Set hardware clock from system time)

sudo hwclock -s #(Set system time from hardware clock)

sudo hwclock -r #(Show hardware clock time)

timedatectl #see all work

systemctl enable systemd-timesyncd.service #auto start systemd-timesyncd when OS boot
